CREATE OR REPLACE TRIGGER emp22_uppercase
BEFORE INSERT OR UPDATE ON emp22
FOR EACH ROW
BEGIN
    :NEW.name := UPPER(:NEW.name);
END;
/




ALTER TABLE emp22
ADD salary NUMBER;


CREATE OR REPLACE TRIGGER emp22_salary_check
BEFORE INSERT OR UPDATE ON emp22
FOR EACH ROW
BEGIN
    IF :NEW.category = 1 AND :NEW.salary <= 10000 THEN
        RAISE_ERROR(-20001, 'Salary must be more than 10000 for category 1');
        
    ELSIF :NEW.category = 2 AND :NEW.salary <= 20000 THEN
        RAISE_ERROR(-20002, 'Salary must be more than 20000 for category 2');
        
    ELSIF :NEW.category = 3 AND :NEW.salary <= 30000 THEN
        RAISE_ERROR(-20003, 'Salary must be more than 30000 for category 3');
    END IF;
END;
/
