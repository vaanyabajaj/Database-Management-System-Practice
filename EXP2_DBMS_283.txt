# Instructor teaching Course
SELECT i.name AS instructor_name,
       c.title AS course_title
FROM instructor i
JOIN teaches t ON i.instructor_id = t.instructor_id
JOIN section s ON t.section_id = s.section_id
JOIN course c ON s.course_id = c.course_id;
